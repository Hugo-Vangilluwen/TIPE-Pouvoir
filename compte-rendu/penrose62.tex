\section{Les lois de Penrose}


\subsection{Système de vote égalitaire}

\begin{definition}
	Un système de vote est dit \textit{égalitaire} si tous les votants ont un poids de 1 et si le quota relatif est fixé à $\nicefrac{1}{2}$. On a alors $W = N$.
\end{definition}

\begin{proposition}
	Dans un système égalitaire,
	\begin{equation}
		\forall i \in \V, \
		B_i = \frac{1}{N} \text{ et }
		B_i' \underset{N \rightarrow +\infty}{\sim} 2\sqrt{\frac{2}{\pi}} \frac{1}{\sqrt{N}}
	\end{equation}
\end{proposition}

\begin{proof}
	Soit $i \in \V$.
	Soit $C \subset \V$ tel que $C \notin \G$ et $C \cup \{i\} \in \G$. \\
	Comme $q = \nicefrac{N}{2}$, on a $\sum_{j \in C} 1 < \nicefrac{N}{2} \leqslant \sum_{j \in C} 1 + 1$. Donc $\nicefrac{N}{2} - 1 \leqslant |C| \leqslant \nicefrac{N}{2}$ i. e. $|C| = \lfloor \nicefrac{N}{2} \rfloor$. D'où :
	\begin{equation*}
		d_i = \binom{N - 1}{\lfloor \nicefrac{N}{2} \rfloor}
	\end{equation*}
	Ainsi $\sum_{i \in \V} d_i = N \binom{N - 1}{\lfloor \nicefrac{N}{2} \rfloor}$ i. e. $B_i = \nicefrac{1}{N}$ et $B_i' = \nicefrac{\binom{N - 1}{\lfloor \nicefrac{N}{2} \rfloor}}{2^{N-1}}$.
	D'après la formule de Stirling,
	\begin{align*}
		B_i'
		&\underset{N \rightarrow +\infty}{\sim} \frac{\binom{N - 1}{\nicefrac{N}{2}}}{2^{N-1}} \\
		&\sim  \frac{ \sqrt{2\pi(N-1)} (N-1)^N }{ 2^{N-1} \sqrt{2\pi\nicefrac{N}{2}} \left(\frac{\nicefrac{N}{2}}{\mathrm{e}}\right)^{\nicefrac{N}{2}} \sqrt{2\pi(N-1-\nicefrac{N}{2})} \left(\frac{N-1-\nicefrac{N}{2}}{\mathrm{e}}\right)^{N-1-\nicefrac{N}{2}} } \\
		&\sim \frac{ \sqrt{N} N^N }{ 2^{N-1} \sqrt{2\pi} \sqrt{\nicefrac{N}{2}} \sqrt{\nicefrac{N}{2}-1} \left(\nicefrac{N}{2}\right)^{\nicefrac{N}{2}} \left(\nicefrac{N}{2}-1\right)^{\nicefrac{N}{2}-1} } \\
		&\sim \frac{ \sqrt{N} N^N }{ 2^{N-1} \sqrt{2\pi} \nicefrac{N}{2} \left(\nicefrac{N}{2}\right)^N } \\
		&\sim \frac{1}{ 2^{-2} \sqrt{2\pi} \sqrt{N}} \\
		&\sim \frac{2\sqrt2}{\sqrt{N\pi}}
	\end{align*}
\end{proof}

Nous retrouvons la loi de la racine carrée de Penrose \cite{penrose62} : pour compenser cette proportionnalité à $\nicefrac{1}{\sqrt{N}}$ et donner à chaque citoyen européen le même pouvoir, il faut que le pouvoir de chaque état membre soit proportionnel à la racine carrée de sa population.
Ce qui contredit l'intuition commune donnant à chaque pays un nombre de sièges proportionnel à sa population.

\begin{proposition}
	Dans un système égalitaire, avec $(\V, \preccurlyeq) = ([\![1;N]\!], \leqslant)$, on a
	\begin{equation}
		\forall i \in \V, \
		B_i^r \underset{N \rightarrow +\infty}{\sim} \left\{ \begin{matrix}
			\frac{1}{2} i N^2 & \text{si } i \leqslant \lceil\nicefrac{N}{2}\rceil \\
			\frac{1}{2} (N - i + 1) N^2 & \text{si } i \geqslant \lceil\nicefrac{N}{2}\rceil \\
		\end{matrix} \right.
		\text{ et }
		B_i'^r = \left\{ \begin{matrix}
			\displaystyle \frac{1}{N - i + 1} & \text{si } i \leqslant \lceil\nicefrac{N}{2}\rceil \\
			~\\
			\displaystyle \frac{1}{i} & \text{si } i \geqslant \lceil\nicefrac{N}{2}\rceil \\
		\end{matrix} \right.
	\end{equation}
\end{proposition}

\begin{proof}
	Soit $i \in \V$. \\
	Soit $C \in \G_i \cap \R$ telle que $C \setminus \{i\} \notin \G$. \\
	On a $\R = \{\emptyset\} \cup \{ [\![i;j]\!] \;|\; (i, j) \in \V^2, i \leqslant j \}$. Donc
	\begin{align*}
		d_i^r
		&= \left| \left\{ [\![j;j']\!] \;|\; (j, j') \in \V^2, j \leqslant j' \text{ et } |j' - j + 1| = \lfloor \nicefrac{N}{2} \rfloor + 1 \text{ et } i \in [\![j;j']\!] \right\} \right| \\
		&= \left| \left\{ [\![ j ; j + \lfloor\nicefrac{N}{2}\rfloor ]\!] \;|\; j \in \mathbb{N}, j \geqslant 1 \text{ et } j + \lfloor\nicefrac{N}{2}\rfloor \leqslant N \text{ et } j \leqslant i \leqslant j + \lfloor\nicefrac{N}{2}\rfloor \right\} \right| \\
		&= \left\{ \begin{matrix}
			| [\![1;i]\!] | & \text{si } i \leqslant \lceil\nicefrac{N}{2}\rceil \\
			| [\![i;N]\!] | & \text{si } i \geqslant \lceil\nicefrac{N}{2}\rceil \\
		\end{matrix} \right. \\
		&= \left\{ \begin{matrix}
			i & \text{si } i \leqslant \lceil\nicefrac{N}{2}\rceil \\
			N - i + 1 & \text{si } i \geqslant \lceil\nicefrac{N}{2}\rceil \\
		\end{matrix} \right. \\
	\end{align*}
	Ainsi
	\begin{align*}
		\sum_{i \in \V} d_i^r
		&= \sum_{i = 1}^{\lceil\nicefrac{N}{2}\rceil} i + \sum_{i = \lceil\nicefrac{N}{2}\rceil + 1}^{N} N - i + 1 \\
		&= \frac{\lceil\nicefrac{N}{2}\rceil \left( \lceil\nicefrac{N}{2}\rceil + 1 \right)}{2} + (N + 1) \left(N - \lceil\nicefrac{N}{2}\rceil + 1\right) - \lceil\nicefrac{N}{2}\rceil \left( N - \lceil\nicefrac{N}{2}\rceil + 1\right) - \sum_{i = 1}^{N - \lceil\nicefrac{N}{2}\rceil} i \\
		&= \frac{\lceil\nicefrac{N}{2}\rceil \left( \lceil\nicefrac{N}{2}\rceil + 1 \right)}{2} + \left(N-\lceil\nicefrac{N}{2}\rceil+1\right)^2 - \frac{\left(N-\lceil\nicefrac{N}{2}\rceil\right) \left(N-\lceil\nicefrac{N}{2}\rceil+1\right)}{2} \\
		&= \left(N-\lceil\nicefrac{N}{2}\rceil+1\right)^2 + \frac{1}{2} \left( - N^2 + 2 N \lceil\nicefrac{N}{2}\rceil + N + 2 \lceil\nicefrac{N}{2}\rceil \right) \\
		&\underset{N \rightarrow +\infty}{\sim} \frac{1}{2} N^2
	\end{align*}
\end{proof}

\subsection{Justification du 62\%}

D'après \cite{votingEU}, il faut choisir un quota relatif qui minimise l'écart entre le pourcentage de sièges et l'indice de Banzhaf absolu soit qui minimise la quantité
\begin{equation*}
	\sigma = \sum_{i \in \V} \left( \frac{w_i}{W} - B_i' \right)^2
\end{equation*}

La répartition des sièges est celle du tableau ci-dessus c'est-à-dire proportionnelle à la racine carrée de la population.

Pour $q_f = 61\%$, on a $\sigma = 3,23 * 10^{-7}$ et pour $q_f = 62\%$, $\sigma = 4,36 * 10^{-7}$. Nous avons rajouté deux pays par rapport à \cite{votingEU}. C'est pourquoi on trouve un quota relatif meilleur à $61\%$. On devrai plutôt appliquer P-61 dans l'Europe des 27.
