\section{Formalisation Mathématiques}

Nous utiliserons les notations de \textit{la mesure du pouvoir de vote} \cite{mesure_pouvoir}. Notons $\V$ l'ensemble des électeurs ou votants. Nous prendrons souvent $\V = [\![1;N]\!]$ avec $N \in \mathbb{N}^*$.


\subsection{Jeu de vote pondéré}

\begin{definition}
	Un \textit{jeu de vote pondéré} est un vote où chaque électeur ou votant ne peut voter que "pour" ou "contre" la proposition. C'est un couple 
	\begin{equation}
		\left( q, \left(w_i\right)_{i \in \V} \right)
		\in \mathbb{N} \times \mathbb{N}^\V
	\end{equation}
	où $q$ est le \textit{quota} nécessaire pour que la proposition soit acceptée et $\left(w_i\right)_{i \in \V}$ est le poids de chaque électeur dans ce vote.
	Posons $W = \displaystyle \sum_{i \in \V} w_i$ la somme de tous les poids. Le jeu de vote pondéré peut aussi être représenté par $(q_f, \left(w_i\right)_{i \in \V}) \in [0;1] \times \mathbb{N}^\V$ où $q_f$ est le quota relatif et $q = \lceil q_f W \rceil$.
\end{definition}

\begin{definition}
	Un \textit{coalition} est dite \textit{gagnante} si
	\begin{equation}
		\sum_{i \in S} w_i \geqslant q
	\end{equation}
	c'est-à-dire si elle peut faire accepter une proposition et dite \textit{perdante} sinon.
	Notons $\G$ l'ensemble des coalitions gagnantes et $\G_i$ l'ensemble de celles contenant l'électeur $i$.
	\begin{equation}
		\G = \left\{ S \in \mathcal{P}(\V) \;|\; \sum_{i \in S} w_i \geqslant q \right\}
	\end{equation}
\end{definition}

\begin{exemple}
	Considérons le bureau d'une association composé d'un président ayant 3 voix, d'un trésorier ayant deux voix et de quatre membres ayant chacun une seule voix. Fixons le quota à la moitié des voix soit à 5 voix.
	La coalition \{président, membre 1, membre 2\} est gagnante et \{trésorier, membre 2, membre 4\} perdante.
\end{exemple}

Une généralisation appelée "jeu coopératif" dans la littérature des votes pondérés est possible avec la fonction suivante. Nous ne l'étudierons pas.
Posons $v : \mathcal{P}\left(\V\right) \rightarrow \left\{0,1\right\}$ la fonction indicatrice de $\mathcal{P}\left(\G\right)$ :
\begin{equation*}
	v(S) = \left\{ \begin{array}{cl}
		1 &\text{ si } \sum_{i \in S} w_i \geqslant q \\
		0 &\text{ sinon}
	\end{array} \right.
\end{equation*}
Un jeu de vote pondéré peut être représenté par $(\V, v)$ (bien que certaines fonction $v : \mathcal{P}\left(\V\right) \rightarrow \left\{0,1\right\}$ ne représente aucun jeu de vote pondéré).


\subsection{Indice de pouvoir}

Le but des indices de pouvoir est de mesurer la capacité de chaque électeur à modifier le résultat du vote. Il en existe des dizaines et nous nous intéresserons à ceux créés par John Banzhaf. Nous utiliserons ici les définitions données dans \textit{La mesure du pouvoir de vote} \cite{mesure_pouvoir}.

\begin{definition}
	Un électeur est dit \textit{décisif} dans une coalition gagnante si lorsque cet électeur se retire de la coalition, celle-ci devient perdante. Notons la décidabilité ou l'indice de Banzhaf brut $d_i$ le nombre de coalition où l'électeur $i$ est décisif.
	\begin{equation}
		d_i
		% = \sum_{S \subset \V, \ i \in S} v(S) - v(S \setminus \{i\})
		= \# \left\{ S \in \G \;|\; i \in S \text{ et } S \setminus \{i\} \notin \G \right\}
	\end{equation}
\end{definition}

\begin{definition}
	L'indice de Banzhaf normalisé $B_i$ est définit par
	\begin{equation}
		B_i = \frac{d_i}{\displaystyle \sum_{j=1}^{n} d_j}
	\end{equation}
	Il est dit normalisé car sa valeur est toujours entre 0 et 1 et la somme des indices de tous les individus vaut 1.
	
	L'indice de Banzhaf relatif $B_i'$ est définit par
	\begin{equation}
		B_i' = \frac{d_i}{\displaystyle 2^{|\V| - 1}}
	\end{equation}
	Cette indice donne la probabilité que le votant $i$ soit dans une coalition gagnante en considérant que ce votant $i$ a une probabilité uniforme d'appartenir à une coalition. En effet, $\# \left\{ S \subset \V \;|\; i \in S \right\} = 2^{|\V| - 1}$.
\end{definition}

\begin{exemple}
	Reprenons l'exemple précédent et calculons les indices de Banzhaf.
	\begin{center}
		\begin{tabular}{|l|l|l|l|}
		\hline
		& brut & normalisé & relatif \\
		\hline
		Président & 11 & 0,44 & 0,6875 \\
		\hline
		Trésorier & 5 & 0,2 & 0,3125 \\
		\hline
		Membres & 3 & 0,12 & 0,1875 \\
		\hline
	\end{tabular}
	\end{center}
\end{exemple}


\subsection{Algorithme de calcul}

\begin{algorithm}[H]
	\caption{Calcul de l'indice de Banzhaf normalisé}
	\Entree{Décidabilité des votants $D = \left(d_v\right)_{v \in \V}$}
	\Sortie{Indice de Branzhaf des votants}
	$sommeD \gets \sum_{v \in \V} d_v$ \quad \emph{Somme des décidabilités} \\
	\uSi{$sommeD = 0$}{\Retour{$(0)_{v \in \V}$}}
	\Sinon{\Retour{$(\nicefrac{d_v}{sommeD})_{v \in \V}$}}
\end{algorithm}

Ce programme a une complexité en $O(N)$ ce qui est inférieur à celle du calcul de la décidabilité.

Commençons par écrire l'algorithme naïf de l'indice de Banzhaf en transcrivant la formule mathématique.
Pour cela, il est nécessaire de calculer $\mathcal{P}(\V)$.

\begin{algorithm}[H]
	\caption{Calcul de l'ensemble des parties d'un ensemble}
	\Entree{Un ensemble E (sous forme de liste)}
	\Sortie{Ensemble des parties}
	\lSi{E est vide}{ \Retour{$\{\emptyset\}$} }
	\lSi{E ne contient qu'un seul élément}{ \Retour{$\{\emptyset, E\}$} }
	Choisir $e \in E$ \\
	
$F \gets Parties(\{ E \setminus \{e\} \})$ \quad \emph{Appel récursif} \\
	$P \gets \emptyset$ \\
	\Pour{$f \in F$}{
		Ajouter à $P$ les ensembles $f$ et $f \cup \{e\}$
	}
	\Retour{P}
\end{algorithm}

En notant $c_n$ la complexité de l'algorithme 1 pour en ensemble de cardinal $n$, nous avons $c_0 = 1$, $c_1 = 2$ et $\forall n \geqslant 1, c_{n+1} = c_n + \Theta(2^n)$.
Or $c_n - c_2 = \sum_{k=2}^{n-1} c_{k+1} - c_k$.
Finalement, la complexité temporelle du calcul de l'ensemble des parties d'un ensemble est en $\Theta(n 2^n)$.

\begin{algorithm}[H]
	\caption{Calcul naïf de la décidabilité}
	\Entree{Un jeu de vote pondéré $(q, (w_v)_{v \in \V})$}
	\Sortie{Décidabilité de tous les votants}
	$G \gets \emptyset$ \quad \emph{Ensemble des parties gagnantes} \\
	\Pour{$S \in Parties(\V)$}{
		$somme \gets \sum_{v \in S} w_v$ \\
		\lSi{$somme \geqslant q$}{ Ajouter à $G$ le tuple $(S, somme)$ }
	}
	$D = (d_v)_{v \in \V} \gets (0)_{v \in \V}$ \quad \emph{Dictionnaire associant à chaque électeur sa décidabilité} \\
	\Pour{$(S, somme) \in G$}{
		\Pour{$v \in S$}{
			\lSi{$(somme - w_v) < q$}{$d_v \gets d_v + 1$}
		}
	}
	\Retour{D}
\end{algorithm}

\begin{proposition}
	Le calcul naïf de l'indice de Banzhaf brut a une complexité temporelle de $O(N 2^N)$.
\end{proposition}

L'appel à $Parties(\V)$ se fait en $\Theta(N 2^N)$.
La première boucle itère sur $2^N$ éléments et effectue un corps de boucle en $O(N)$ (car $|S| \leqslant |\V|$) soit une complexité de $O(N 2^N)$. La seconde boucle a exactement la même complexité.


\subsection{Polynôme générateur de Brams-Affuso}

Pour diminuer cette complexité, nous allons utiliser le polynôme générateur de Brams-Affuso \cite{power_european_constitution}.

Notons $c_k^i$ le nombre de coalitions n'incluant pas $i$ et de poids total $k$ (somme des poids des votants qu'elles contiennent).
Pour que le votant $i$ soit décisif dans une coalition $S$, il faut que $S \setminus \{i\}$ soit perdante mais de poids total pas trop faible de sorte que l'ajout de $i$ la rende gagnante, le poids total de $S \setminus \{i\}$ doit donc être compris entre $q - w_i$ et $q - 1$. Formellement avec $S \in \mathcal{P}(\V)$ :
\begin{equation*}
	\left\{ \begin{matrix}
		i \in S \\
		S \in \G \\
		S \setminus \{i\} \not\in \G
	\end{matrix} \right.
	\iff
	\left\{ \begin{matrix}
		i \in S \\
		\sum_{j \in S} w_j = \sum_{j \in S \setminus \{i\}} w_j + w_i \geqslant q \\
		\sum_{j \in S \setminus \{i\}} w_j < q
	\end{matrix} \right.
	\iff
	\left\{ \begin{matrix}
		i \in S \\
		q - w_i \leqslant \sum_{j \in S \setminus \{i\}} w_j \leqslant q - 1
	\end{matrix} \right.
\end{equation*}
On a donc l'égalité ensembliste :
\begin{equation*}
	\left\{ S \in \G \;|\; i \in S \text{ et } S \setminus \{i\} \notin \G \right\}
	= \left\{ S \in \mathcal{P}(\V) \;|\; i \not\in S \text{ et } q - w_i \leqslant \sum_{j \in S} w_j \leqslant q - 1 \right\}
\end{equation*}

Or les poids des votants sont des nombres entiers.
Nous obtenons donc la formule ci-dessous pour tout $i \in \V$.
\begin{equation}
	d_i = \sum_{k=\max(0, q-w_i)}^{q-1} c_k^i
\end{equation}

\begin{proposition}
	Brams et Affuso ont démontré l'égalité suivante :
	\begin{equation}
		\prod_{j \in \V, \; j \neq i} \left(1 + X^{w_j}\right)
		= \sum_{k=0}^{W-w_i} c_k^i X^k
	\end{equation}
	On l'appelle polynôme générateur de Brams-Affuso (de la suite $(c_k^i)_{k \in \mathbb{N}}$).
\end{proposition}

\begin{proof}
	Nous n'avons pas trouvé leur démonstration et l'avons prouvé de notre coté.
	Remarquons que le degré du polynôme de gauche est bien $\displaystyle \sum_{j \in \V, j \neq i} w_j  = W - w_i$.
	Développons ce produit :
	\begin{equation*}
		\prod_{\footnotesize \begin{matrix} j \in \V \\ j \neq i \end{matrix}} \left(1 + 1 . X^{w_j}\right)
		= \sum_{k = 0}^{W - w_i} \left( \sum_{\footnotesize \begin{matrix} S \in \mathcal{P}\left(\V \setminus \{i\}\right) \\ \sum_{j \in S} w_j = k \end{matrix}} 1 \right) X^k
	\end{equation*}
	
	L'expression des coefficients polynomiaux à droite est exactement celle des $c_k^i$.
\end{proof}


Nous avons implémenté en python les polynômes sous forme de classe en les munissant des opérations d'addition, de multiplication par un scalaire et de multiplication par un polynôme. $P + Q$ se calcule en $\Theta(\min(\deg P, \deg Q))$, $\lambda . P$ en $\Theta(\deg P)$ et $P \times Q$ en $O(\min(\deg P, \deg Q) \times (\deg P + \deg Q))$.

\begin{algorithm}[H]
	\caption{Calcul de la décidabilité avec les polynômes générateurs de Brams-Affuso}
	\Entree{Un jeu de vote pondéré $(q, (w_v)_{v \in \V})$}
	\Sortie{Décidabilité de tous les votants}
	$D = (d_v)_{v \in \V} \gets (0)_{v \in \V}$ \quad \emph{Dictionnaire associant à chaque électeur sa décidabilité} \\
	\Pour{$v \in \V$}{
		$P \gets 1$ \quad \emph{Polynôme générateur} \\
		\Pour{$u \in \V$}{
			\lSi{$v \neq u$}{$P \gets P \times (1 + X^{w_u})$}
		}
		\Pour{$i$ de $\max(0, q - w_v)$ à $q - 1$}{
			$d_v \gets d_v + \text{coefficient de }X^i\text{ dans }P$
		}
	}
	\Retour{D}
\end{algorithm}

\begin{proposition}
	La complexité temporelle de cet algorithme est $O(N . W^2)$.
\end{proposition}

\begin{proof}
	L'initialisation s'effectue en $O(W)$ car $|\V| \leqslant W$.
	$P$ a toujours un degré inférieur à $W$. Donc $P \times (1 + X^{w_u})$ se calcule en $O(w_u \times (W + w_u))$.
	Or $w_u \leqslant W$ donc ce produit se fait en $O(W w_u)$.
	La première boucle s'effectue en $\sum_{u \in \V} O(W w_u) = O(W^2)$.
	Pour la deuxième boucle, on a $\# [\![\max(0,q-w_v);q-1]\!] \leqslant q$ et on peut considérer que $q \leqslant W$ (sinon le jeu de vote pondéré étudié n'a aucun intérêt).
	La boucle principale s'effectue donc en $O(|\V| W^2)$.
\end{proof}


\subsection{Problème inverse}

Nous pouvons aussi prendre le problème dans le sens inverse.
Jusqu'ici, nous calculions l'indice de Banzhaf connaissant les poids des électeur. Nous allons chercher les poids des électeurs à partir d'indices de Banzhaf relatifs donnés (ou désirés) noté $d \in \mathbb{N}^\V$ et du quota $q \in \mathbb{N}$.
Dans \cite{algorithm_weight}, Dennis Leech propose de le résoudre le problème de manière itérative avec la suite suivante :
\begin{equation}
	p_0 = d
\end{equation}
\begin{equation}
	\forall n \in \mathbb{N}, \
	p_{n+1} = \lambda(n) \times \left( d - p_n \right)
\end{equation}

Pour l'implémentation de notre algorithme nous avons choisi d'utiliser :
\begin{equation*}
	\lambda(n) = \frac{0,\!83 q}{\ln(n)+1}
\end{equation*}
Nous effectuons 40 itérations.

\section{Coalitions réalistes}
Dans cette partie, nous munissons les votants d'une relation d'ordre "est plus à gauche que". Nous considérons donc $(\V, \preccurlyeq)$.
Nous avons créé la notion de coalitions réalistes.

\begin{definition}
	Une coalition $C \in \mathcal{P}(\V)$ est dite \textit{réaliste} si
	\begin{equation}
		\forall (u, v) \in \V^2, \forall w \in \V,
		u \preccurlyeq w \preccurlyeq v \implies w \in C
	\end{equation}
	Notons $\mathcal{R}$ l'ensemble des coalitions réalistes.
\end{definition}

Nous pouvons donc redéfinir la décidabilité et les indices de Banzhaf.

\begin{definition}
	Posons $d_i^r$ la décidabilité réaliste d'un électeur $i$ :
	\begin{equation}
		d_i^r
		= \# \left\{ S \in \G \cap \mathcal{R} \;|\; i \in S \text{ et } S \setminus \{i\} \notin \G \right\}
	\end{equation}
	De même, posons les indices de Banzhaf réalistes normalisé et relatif :
	\begin{equation*}
		B_i = \frac{d_i^r}{\displaystyle \sum_{j=1}^{n} d_j^r}
		\qquad \qquad
		B_i' = \frac{d_i^r}{\displaystyle 2^{N - 1}}
	\end{equation*}
\end{definition}

\begin{algorithm}[H]
	\caption{Calcul de l'ensemble des coalitions réalistes de $(E, \preccurlyeq)$}
	\Entree{Une liste L des éléments de E triée selon $\preccurlyeq$}
	\Sortie{Ensemble des coalitions réalistes}
	
	$N \gets Longueur(L)$ \\
	$R \gets \{ \emptyset \}$ \\
	
	\Pour{$i$ de $1$ à $N$}{
		\Pour{$j$ de $i+1$ à $N$}{
			Ajouter à $R$ la sous-liste de $L$ entre i et j inclus
		}
	}
	\Retour{R}
\end{algorithm}

\begin{proposition}
	L'algorithme naïf de la décidabilité réaliste (celui naïf de la décidabilité où l'on remplace $Parties(\V)$ par $Réalistes(\V, \preccurlyeq)$) a une complexité temporelle de $O(N^3)$.
\end{proposition}

\begin{proof}
	La complexité temporelle de l'extraction d'une sous-liste est linéaire en la taille de la liste. Donc l'algorithme calculant l'ensemble des coalitions réalistes a une complexité en $\Theta(N^3)$.
	
	De plus, d'après cet algorithme, $\# \mathcal{R} = 1 + \sum_{i=1}^{N} (N - i) = 1 + N^2 - \frac{N^2 + N}{2} = 1 + \frac{N^2 - N}{2}$ d'où $\# \mathcal{R} \leqslant N^2$.
	Dans l'algorithme 3, le corps des deux boucles se font en $O(N)$.
	Ainsi la complexité de l'algorithme naïf de la décidabilité réaliste est en $O(N^3)$.
\end{proof}
