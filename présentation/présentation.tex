	\subsection{Jeu de vote pondéré}
	
	\begin{frame}
		\begin{definition}
			\textbf{Jeu de vote pondéré} : Pour / Contre \\
			$\mathcal{V}$ ensemble des votants. Généralement, $\mathcal{V} = [\![ 1; N ]\!]$. \\
			\begin{equation*}
				(q, \left(w_i\right)_{i \in \mathcal{V}})
				\in \mathbb{N} \times \mathbb{N}^\mathcal{V}
				% quota, poids
			\end{equation*}
			Poids total $W = \sum_{i \in \mathcal{V}} w_i$ \\
			\textcolor{red}{Quota relatif $q_f \in [0;1]$ tel que $q = \lceil q_f W  \rceil$}
		\end{definition}
		\uncover<2>{\begin{definition}
				\textbf{Coalition gagnante} $C \subset \mathcal{V}$
				\begin{equation*}
					\sum_{i \in C} w_i \geqslant q
				\end{equation*}
				% sinon perdante
				Notons $\mathcal{G} \subset \mathcal{P}(\mathcal{V})$ l'ensemble des coalitions gagnantes.
		\end{definition}}
	\end{frame}
	
	\begin{frame}
		\begin{definition}
			Un électeur est dit \textbf{décisif} pour une coalition gagnante si le retrait de celui-ci la rend perdante. \\
			\uncover<2>{
				La décidabilité $d_i$ pour $i \in \mathcal{V}$ est le nombre de coalitions où $i$ est décisif :
				\begin{equation*}
					d_i = \left| \left\{ C \in \mathcal{G} \;|\; i \in C \text{ et } C \setminus \{i\} \notin \mathcal{G} \right\} \right|
				\end{equation*}
			}
		\end{definition}
	\end{frame}
	
	\subsection{Indice de Banzhaf}
	\begin{frame}{Indice de Banzhaf}
		\begin{definition}
			Indice de Banzhaf normalisé :
			\begin{equation*}
				B_i = \frac{d_i}{\sum_{j=1}^{N} d_j}
			\end{equation*}
			Indice de Banzhaf absolu :
			\begin{equation*}
				B_i' = \frac{d_i}{\displaystyle 2^{N - 1}}
			\end{equation*}
		\end{definition}
	\end{frame}
	
	\begin{frame}{Algorithme naïf}

		Complexité : $\mathcal{O}(N 2^N)$ \\

	\subsection{Polynôme générateur de Brams-Affuso}
	
	\begin{frame}{Polynôme générateur de Brams-Affuso}
		$d_i$ : décidabilité du votant $i$ \\
		$c_k^i$ : nombre de coalitions n'incluant pas $i$ et de poids total $k$
		\color{red}
		\uncover<2-4>{
			\begin{equation*}
				\left\{ \begin{matrix}
					i \in C \\
					C \in \mathcal{G} \\
					C \setminus \{i\} \not\in \mathcal{G}
				\end{matrix} \right.
				\iff
				\left\{ \begin{matrix}
				i \in C \\
				\sum_{j \in C} w_j = \sum_{j \in C \setminus \{i\}} w_j + w_i \geqslant q \\
				\sum_{j \in C \setminus \{i\}} w_j < q
				\end{matrix} \right.
			\end{equation*}
		}
		\uncover<3-4>{
			\begin{equation*}
				\iff
				\left\{ \begin{matrix}
					i \in C \\
					q - w_i \leqslant \sum_{j \in C \setminus \{i\}} w_j \leqslant q - 1
				\end{matrix} \right.
			\end{equation*}
		}
		\uncover<4>{
			\begin{equation*}
				d_i = \sum_{k=q-w_i}^{q-1} c_k^i
			\end{equation*}
		}
	\end{frame}
	
	\begin{frame}{Polynôme générateur de Brams-Affuso}
		\begin{equation*}
				\prod_{j \in \mathcal{V}, \; j \neq i} \left(1 + X^{w_j}\right)
				\alt<3>{=}{\overset{?}{=}}
				\sum_{k=0}^{W-w_i} c_k^i X^k
		\end{equation*}
		\color{red}
		\uncover<2->{\begin{equation*}
			\prod_{\footnotesize \begin{matrix} j \in \mathcal{V} \\ j \neq i \end{matrix}} \left(1 + 1 . X^{w_j}\right)
			= \sum_{k = 0}^{W - w_i} \left( \sum_{\footnotesize \begin{matrix} C \in \mathcal{P}\left(\mathcal{V} \setminus \{i\}\right) \\ \sum_{j \in C} w_j = k \end{matrix}} 1 \right) X^k
		\end{equation*}}
		\uncover<3>{Complexité : $\mathcal{O}(N . W^2)$}
	\end{frame}

	\begin{frame}{Jeu de vote égalitaire}
		\color{red}
		Si $w = (1)_{i \in \mathcal{V}}$ et $q_f = \nicefrac{1}{2}$ alors
		\uncover<2->{
			\begin{equation*}
				\forall i \in \mathcal{V}, \
				d_i = \binom{N - 1}{\lfloor \nicefrac{N}{2} \rfloor}
			\end{equation*}
		}
		\uncover<3>{
			\begin{equation*}
				\forall i \in \mathcal{V}, \
				B_i = \frac{1}{N} \text{ et }
				B_i' \underset{N \rightarrow +\infty}{\sim} 2\sqrt{\frac{2}{\pi}} \frac{1}{\sqrt{N}}
			\end{equation*}
		}

	\end{frame}
	
	\begin{frame}{Théorème de Penrose}
		\begin{theorem}
			Tous les participants d’un jeu en 2 étapes ont le même pouvoir si les indices de pouvoir des groupes de la $2^{\grave{e}me}$ étape sont proportionnels aux racines carrées de leur cardinal.
		\end{theorem}
		\uncover<2>{
			\medbreak
			\underline{Pour l’UE} : tous les citoyens possèdent le même pouvoir si les indices de pouvoirs des pays au parlement européen sont proportionnels à la racine carrée de leur population.
		}
	\end{frame}

	\subsection{Coalitions réalistes}
	
	\begin{frame}{Coalitions réalistes}
		\color{red}
		\begin{equation*}
			(\mathcal{V}, \preccurlyeq)
		\end{equation*}
		
		\uncover<2>{
			\begin{definition}
				\color{red}
				Une coalition $C \in \mathcal{P}(\mathcal{V})$ est dite \textit{réaliste} si
				\begin{equation*}
					\forall (i, j) \in C^2, \forall k \in \mathcal{V},
					i \preccurlyeq k \preccurlyeq j \implies k \in C
				\end{equation*}
				Notons $\mathcal{R}$ l'ensemble des coalitions réalistes et $\mathcal{R}_i$ celles contenant le votant $i$.
			\end{definition}
		}
	\end{frame}
	
	\begin{frame}
		\color{red}
		$\mathcal{V} = \{G, C, D\}$ avec $G \preccurlyeq C \preccurlyeq D$
		
		\begin{equation*}
			\begin{tikzcd}[column sep=0, row sep=1.2cm, ampersand replacement=\&]
				\&\& \{G, C, D\} \drar[dash] \dlar[dash] \&\& \\
				\& \{G, C\} \drar[dash] \dlar[dash] \&\& \{C, D\} \drar[dash] \dlar[dash] \& \\
				\{G\} \arrow[dash]{drr} \&\& \{C\} \dar[dash] \&\& \{D\} \arrow[dash]{dll} \\
				\&\& \emptyset \&\& \\
			\end{tikzcd}
		\end{equation*}
		\vspace{-1.9cm}
		\begin{center}
			Treillis $(\mathcal{R}, \subseteq)$
		\end{center}
	\end{frame}
	
	\begin{frame}{Décidabilité réaliste}
		\color{red}
		\begin{equation*}
			d_i^r
			= \left| \left\{ C \in \mathcal{G} \cap \mathcal{R}_i \;|\; C \setminus \{i\} \notin \mathcal{G} \right\} \right|
		\end{equation*}
		~\\
		\uncover<2>{\begin{equation*}
			B_i^r = \frac{d_i^r}{\displaystyle \sum_{j=1}^{n} d_j^r}
			\qquad \qquad
			B_i'^r = \frac{d_i^r}{\displaystyle |\mathcal{R}_i|}
		\end{equation*}}
	\end{frame}
	
	\begin{frame}
		\color{red}
		Considérons $\left( \mathcal{V}, \preccurlyeq \right) = \left( [\![1; N]\!], \leqslant \right)$.
		\begin{equation*}
			\mathcal{R} = \left\{\emptyset\right\} \cup \left\{ [\![j;j']\!] \;|\; (j, j') \in \mathcal{V}^2, j \leqslant j' \right\}
		\end{equation*}
		\uncover<2>{Complexité de $d_i^r$ : $\mathcal{O}(N^3)$}
	\end{frame}
