	\subsection{Jeu de vote pondéré}
	\begin{frame}
		\begin{definition}
			\textbf{Jeu de vote pondéré} : Pour / Contre \\
			\begin{equation*}
				(q, \left(w_i\right)_{i \in \mathcal{V}})
				\in \mathbb{N} \times \mathbb{N}^\mathcal{V}
				% quota, poids
			\end{equation*}
			$\mathcal{V}$ ensemble des votants. Généralement, $\mathcal{V} = [\![ 1; N ]\!]$. \\
			Poids total : $W = \sum_{i \in \mathcal{V}} w_i$
		\end{definition}
		\uncover<2>{\begin{definition}
				\textbf{Coalition gagnante} $S$
				\begin{equation*}
					\sum_{i \in S} w_i \geqslant q
				\end{equation*}
				% sinon perdante
				Notons $\mathcal{G} \subset \mathcal{P}(\mathcal{V})$ l'ensemble des coalitions gagnantes.
		\end{definition}}
	\end{frame}
	
	\begin{frame}
		\begin{definition}
			Un électeur est dit \textbf{décisif} pour un coalition gagnante si le retrait de celui-ci la rend perdante. \\
			\uncover<2>{
				La décidabilité $d_i$ pour $i \in \mathcal{V}$ est le nombre de coalitions où $i$ est décisif :
				\begin{equation*}
					d_i =
					\# \left\{ S \in \mathcal{G} \;|\; i \in S \text{ et } S \setminus \{i\} \notin \mathcal{G} \right\}
				\end{equation*}
			}
		\end{definition}
	\end{frame}
	
	\subsection{Indice de Banzhaf}
	\begin{frame}
		\begin{definition}
			Indice de Banzhaf normalisé :
			\begin{equation*}
				B_i = \frac{d_i}{\sum_{j=1}^{N} d_j}
			\end{equation*}
			Indice de Banzhaf relatif :
			\begin{equation*}
				B_i' = \frac{d_i}{2^{|\mathcal{V}| - 1}}
			\end{equation*}
		\end{definition}
		\uncover<2>{L'algorithme naïf du calcul de l'indice de Banzhaf a une complexité de $O(N 2^N)$.}
	\end{frame}
	
	\subsection{Polynôme générateur de Brams-Affuso}
	\begin{frame}
		\underline{Polynôme générateur de Brams-Affuso} \\
		\uncover<2-5>{
			$d_i$ : décidabilité du votant $i$ \\
			$c_k^i$ : nombre de coalitions n'incluant pas $i$ et de poids total $k$
		}
		\uncover<3-5>{\begin{equation*}
			\left\{ \begin{matrix}
				i \in S \\
				S \in \mathcal{G} \\
				S \setminus \{i\} \not\in \mathcal{G}
			\end{matrix} \right.
			\iff
			\left\{ \begin{matrix}
			i \in S \\
			\sum_{j \in S} w_j = \sum_{j \in S \setminus \{i\}} w_j + w_i \geqslant q \\
			\sum_{j \in S \setminus \{i\}} w_j < q
			\end{matrix} \right.
		\end{equation*}}
		\uncover<4-5>{\begin{equation*}
			\iff
			\left\{ \begin{matrix}
				i \in S \\
				q - w_i \leqslant \sum_{j \in S \setminus \{i\}} w_j \leqslant q - 1
			\end{matrix} \right.
		\end{equation*}}
		\uncover<5>{\begin{equation*}
				d_i = \sum_{k=q-w_i}^{q-1} c_k^i
		\end{equation*}}
	\end{frame}
	
	\begin{frame}
		\underline{Polynôme générateur de Brams-Affuso} \\
		\begin{equation*}
				\prod_{j \in \mathcal{V}, \; j \neq i} \left(1 + X^{w_j}\right)
				= \sum_{k=0}^{W-w_i} c_k^i X^k
		\end{equation*}
		\uncover<2-3>{\begin{equation*}
			\prod_{\footnotesize \begin{matrix} j \in \mathcal{V} \\ j \neq i \end{matrix}} \left(1 + 1 . X^{w_j}\right)
			= \sum_{k = 0}^{W - w_i} \left( \sum_{\footnotesize \begin{matrix} S \in \mathcal{P}\left(\mathcal{V} \setminus \{i\}\right) \\ \sum_{j \in S} w_j = k \end{matrix}} 1 \right) X^k
		\end{equation*}}
		\uncover<3>{Complexité : $O(N . W^2)$}
	\end{frame}

	\begin{frame}{Problème inverse}
		\underline{Algorithme de Leech}
		\begin{equation*}
			w_0 = d
		\end{equation*}
		\bigbreak
		\begin{equation*}
			w_{n+1} = \lambda(n) ( d - w_n )
		\end{equation*}
		\bigbreak
		\begin{equation*}
			\lambda(n) = \frac{0,\!83 q}{\ln(i)+1}
		\end{equation*}
	\end{frame}
