	\subsection{Jeu de vote pondéré}
	\begin{frame}
		\begin{definition}
			\textbf{Jeu de vote pondéré} : Pour / Contre \\
			\begin{equation*}
				(q, \left(w_i\right)_{i \in \mathcal{V}})
				\in \mathbb{N} \times \mathbb{N}^\mathcal{V}
				% quota, poids
			\end{equation*}
			$\mathcal{V}$ ensemble des votants. Généralement, $\mathcal{V} = [\![ 1; N ]\!]$. \\
			Poids total : $W = \sum_{i \in \mathcal{V}} w_i$ \\
			Posons le quota relatif $q_f \in [0;1]$ tel que $q = \lceil q_f W  \rceil$.
		\end{definition}
		\uncover<2>{\begin{definition}
				\textbf{Coalition gagnante} $C$
				\begin{equation*}
					\sum_{i \in C} w_i \geqslant q
				\end{equation*}
				% sinon perdante
				Notons $\mathcal{G} \subset \mathcal{P}(\mathcal{V})$ l'ensemble des coalitions gagnantes.
		\end{definition}}
	\end{frame}
	
	\begin{frame}
		\begin{definition}
			Un électeur est dit \textbf{décisif} pour un coalition gagnante si le retrait de celui-ci la rend perdante. \\
			\uncover<2>{
				La décidabilité $d_i$ pour $i \in \mathcal{V}$ est le nombre de coalitions où $i$ est décisif :
				\begin{equation*}
					d_i = \left| \left\{ C \in \mathcal{G} \;|\; i \in C \text{ et } C \setminus \{i\} \notin \mathcal{G} \right\} \right|
				\end{equation*}
			}
		\end{definition}
	\end{frame}
	
	\subsection{Indice de Banzhaf}
	\begin{frame}
		\begin{definition}
			Indice de Banzhaf normalisé :
			\begin{equation*}
				B_i = \frac{d_i}{\sum_{j=1}^{N} d_j}
			\end{equation*}
			Indice de Banzhaf absolu :
			\begin{equation*}
				B_i' = \frac{d_i}{2^{N - 1}}
			\end{equation*}
		\end{definition}
		\uncover<2>{L'algorithme naïf du calcul de l'indice de Banzhaf a une complexité de $O(N 2^N)$.}
	\end{frame}
	
	\subsection{Polynôme générateur de Brams-Affuso}
	\begin{frame}
		\underline{Polynôme générateur de Brams-Affuso} \\
		$d_i$ : décidabilité du votant $i$ \\
		$c_k^i$ : nombre de coalitions n'incluant pas $i$ et de poids total $k$
		\uncover<2-4>{\begin{equation*}
			\left\{ \begin{matrix}
				i \in C \\
				C \in \mathcal{G} \\
				C \setminus \{i\} \not\in \mathcal{G}
			\end{matrix} \right.
			\iff
			\left\{ \begin{matrix}
			i \in C \\
			\sum_{j \in C} w_j = \sum_{j \in C \setminus \{i\}} w_j + w_i \geqslant q \\
			\sum_{j \in C \setminus \{i\}} w_j < q
			\end{matrix} \right.
		\end{equation*}}
		\uncover<3-4>{\begin{equation*}
			\iff
			\left\{ \begin{matrix}
				i \in C \\
				q - w_i \leqslant \sum_{j \in C \setminus \{i\}} w_j \leqslant q - 1
			\end{matrix} \right.
		\end{equation*}}
		\uncover<4>{\begin{equation*}
				d_i = \sum_{k=q-w_i}^{q-1} c_k^i
		\end{equation*}}
	\end{frame}
	
	\begin{frame}
		\underline{Polynôme générateur de Brams-Affuso} \\
		\begin{equation*}
				\prod_{j \in \mathcal{V}, \; j \neq i} \left(1 + X^{w_j}\right)
				= \sum_{k=0}^{W-w_i} c_k^i X^k
		\end{equation*}
		\uncover<2-3>{\begin{equation*}
			\prod_{\footnotesize \begin{matrix} j \in \mathcal{V} \\ j \neq i \end{matrix}} \left(1 + 1 . X^{w_j}\right)
			= \sum_{k = 0}^{W - w_i} \left( \sum_{\footnotesize \begin{matrix} C \in \mathcal{P}\left(\mathcal{V} \setminus \{i\}\right) \\ \sum_{j \in C} w_j = k \end{matrix}} 1 \right) X^k
		\end{equation*}}
		\uncover<3>{Complexité : $O(N . W^2)$}
	\end{frame}

	\subsection{Coalitions réalistes}
	
	\begin{frame}
		Considérons maintenant $(\mathcal{V}, \preccurlyeq)$.
		
		\uncover<2>{\begin{definition}
			Une coalition $C \in \mathcal{P}(\mathcal{V})$ est dite \textit{réaliste} si
			\begin{equation*}
				\forall (i, j) \in \mathcal{V}^2, \forall k \in \mathcal{V},
				i \preccurlyeq k \preccurlyeq j \implies k \in C
			\end{equation*}
			Notons $\mathcal{R}$ l'ensemble des coalitions réalistes et $\mathcal{R}_i$ celles contenant le votant $i$.
		\end{definition}}
	\end{frame}
	
	\begin{frame}
		$\mathcal{V} = \{G, C, D\}$ avec $G \preccurlyeq C \preccurlyeq D$
		
		\uncover<2>{\begin{equation*}
			\begin{tikzcd}[column sep=0, row sep=1.2cm, ampersand replacement=\&]
				\&\& \{G, C, D\} \drar[dash] \dlar[dash] \&\& \\
				\& \{G, C\} \drar[dash] \dlar[dash] \&\& \{C, D\} \drar[dash] \dlar[dash] \& \\
				\{G\} \arrow[dash]{drr} \&\& \{C\} \dar[dash] \&\& \{D\} \arrow[dash]{dll} \\
				\&\& \emptyset \&\& \\
			\end{tikzcd}
		\end{equation*}
		\vspace{-1.9cm}
		\begin{center}
			diagramme de Hasse de $(\mathcal{R}, \subseteq)$
		\end{center}}
	\end{frame}
	
	\begin{frame}
		\begin{equation*}
			d_i^r
			= \left| \left\{ C \in \mathcal{G} \cap \mathcal{R}_i \;|\; C \setminus \{i\} \notin \mathcal{G} \right\} \right|
		\end{equation*}
		~\\
		\uncover<2>{\begin{equation*}
			B_i^r = \frac{d_i^r}{\displaystyle \sum_{j=1}^{n} d_j^r}
			\qquad \qquad
			B_i'^r = \frac{d_i^r}{\displaystyle |\mathcal{R}_i|}
		\end{equation*}}
	\end{frame}
	
	\begin{frame}
		Considérons $\left( \mathcal{V}, \preccurlyeq \right) = \left( [\![1; N]\!], \leqslant \right)$.
		\begin{equation*}
			\mathcal{R} = \left\{\emptyset\right\} \cup \left\{ [\![j;j']\!] \;|\; (j, j') \in \mathcal{V}^2, j \leqslant j' \right\}
		\end{equation*}
		\uncover<2>{Complexité : $O(N^3)$}
	\end{frame}
