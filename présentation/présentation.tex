	\section{Théorie du choix social}
	
	\subsection{Jeu de vote pondéré}
	\begin{frame}
		\begin{definition}
			\textbf{Jeu de vote pondéré} : Pour / Contre \\
			\begin{equation*}
				(q, \left(w_i\right)_{i \in \mathcal{V}}) % quota, poids
			\end{equation*}
			$\mathcal{V}$ ensemble des votants. Généralement, $\mathcal{V} = [\![ 1; N ]\!]$.
		\end{definition}
		\uncover<2>{\begin{definition}
				\textbf{Coalition gagnante} $S$
				\begin{equation*}
					\sum_{i \in S} w_i \geqslant q
				\end{equation*}
				% sinon perdante
				Notons $\mathcal{G} \in \mathcal{P}(\mathcal{V})$ l'ensemble des coalitions gagnantes.
		\end{definition}}
	\end{frame}
	
	\begin{frame}
		\begin{definition}
			Un électeur est dit \textbf{décisif} pour un coalition gagnante si le retrait de celui-ci la rend perdante. \\
			\uncover<2>{
				La décidabilité $d_i$ pour $i \in \mathcal{V}$ est le nombre de coalitions où $i$ est décisif :
				\begin{equation*}
					d_i =
					\# \left\{ S \in \mathcal{G} \;|\; i \in S \text{ et } S \setminus \{i\} \notin \mathcal{G} \right\}
				\end{equation*}
			}
		\end{definition}
	\end{frame}
	
	\subsection{Indice de Banzhaf}
	\begin{frame}
		\begin{definition}
			Indice de Banzhaf normalisé :
			\begin{equation*}
				B_i = \frac{d_i}{\sum_{j=1}^{n} d_j}
			\end{equation*}
			Indice de Banzhaf relatif :
			\begin{equation*}
				B_i' = \frac{d_i}{2^{|\mathcal{V}| - 1}}
			\end{equation*}
		\end{definition}
		\uncover<2>{L'algorithme naïf du calcul de l'indice de Banzhaf normalisé a une complexité de $O(n 2^n)$.}
	\end{frame}
	
	\begin{frame}
		\underline{Algorithme de Brams-Affuso} \\
		\uncover<2-5>{
			$d_i$ : décidabilité du votant $i$ \\
			$d_k^i$ : nombre de coalitions n'incluant pas $i$ et de poids total $k$
		}
		\uncover<3-5>{\begin{equation*}
				d_i = \sum_{k=q-w_i}^{q-1} d_k^i
				\qquad \qquad
				W = \sum_{i \in \mathcal{V}} w_i
		\end{equation*}}
		\uncover<4-5>{\begin{equation*}
				\prod_{j \in \mathcal{V}, \; j \neq i} \left(1 + X^{w_j}\right)
				= \sum_{k=0}^{W-w_i} d_k^i X^k
		\end{equation*}}
		\uncover<5>{Complexité : $O(n^3)$}
	\end{frame}

	\begin{frame}{Problème inverse}
		\underline{Algorithme de Leech}
		\begin{equation*}
			w_0 = d
		\end{equation*}
		\bigbreak
		\begin{equation*}
			w_{n+1} = \lambda(n) ( d - w_n )
		\end{equation*}
		\bigbreak
		\begin{equation*}
			\lambda(n) = \frac{0,\!83 q}{\ln(i)+1}
		\end{equation*}
	\end{frame}
